## 初始代码

```cpp
class Solution {
public:
    bool isMonotonic(vector<int>& nums) {
        int zeng=0;int jian=2;int sz=nums.size();int szz=sz-1;
        if (sz==1) return true;
        for (int i=0;i<szz&&zeng!=jian;i++){
            if (nums[i]<nums[i+1]){
                zeng=1;
            }
            else{
                if (nums[i]>nums[i+1]){
                    jian=1;
                }
            }
        }
        if (zeng==jian) return false;
        return true;
    }
};
```
